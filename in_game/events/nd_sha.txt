namespace = nd_sha

# Formation event — fires when the Shan States are formed
nd_sha.1 = {
	type = country_event
	title = nd_sha.1.t
	desc = nd_sha.1.d
	fire_only_once = yes

	option = {
		name = nd_sha.1.a
		add_country_modifier = {
			modifier = nd_confederation_of_the_sword
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_sha_formed
			set_variable = nd_sha_path_a
		}
	}
	option = {
		name = nd_sha.1.b
		add_country_modifier = {
			modifier = nd_crossroads_of_the_plateau
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_sha_formed
			set_variable = nd_sha_path_b
		}
	}
}
