namespace = nd_sar

# Formation event - the founding of the Kingdom of Sardinia
nd_sar.1 = {
	type = country_event
	title = nd_sar.1.t
	desc = nd_sar.1.d
	fire_only_once = yes

	option = {
		name = nd_sar.1.a
		add_country_modifier = {
			modifier = nd_sar_carta_de_logu
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_minor_move_to_right
		}
		hidden_effect = {
			set_variable = nd_sar_formed
			set_variable = nd_sar_path_a
		}
	}
	option = {
		name = nd_sar.1.b
		add_country_modifier = {
			modifier = nd_sar_savoy_ambition
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_minor_move_to_left
		}
		hidden_effect = {
			set_variable = nd_sar_formed
			set_variable = nd_sar_path_b
		}
	}
}
