namespace = nd_hin

# Formation event — fires when Hindustan is formed
nd_hin.1 = {
	type = country_event
	title = nd_hin.1.t
	desc = nd_hin.1.d
	fire_only_once = yes

	option = {
		name = nd_hin.1.a
		add_country_modifier = {
			modifier = nd_padshah_e_hind
			years = 40
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_hin_formed
			set_variable = nd_hin_path_a
		}
	}
	option = {
		name = nd_hin.1.b
		add_country_modifier = {
			modifier = nd_daulat_e_hindustan
			years = 40
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_hin_formed
			set_variable = nd_hin_path_b
		}
	}
	option = {
		name = nd_hin.1.c
		add_country_modifier = {
			modifier = nd_chakravarti_samrat
			years = 40
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_hin_formed
			set_variable = nd_hin_path_c
		}
	}
}
