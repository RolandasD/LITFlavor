namespace = nd_yav

# Formation event — The Yavana Restoration
nd_yav.1 = {
	type = country_event
	title = nd_yav.1.t
	desc = nd_yav.1.d
	fire_only_once = yes

	option = {
		name = nd_yav.1.a
		add_country_modifier = {
			modifier = nd_yav_phalanx_reborn
			years = 30
			mode = add_and_extend
		}
		change_societal_value = {
			type = quality_vs_quantity
			value = societal_value_move_to_left
		}
		change_culture = culture:yavana_culture
		custom_tooltip = nd_yav_culture_conversion_tt
		hidden_effect = {
			set_variable = nd_yav_formed
			set_variable = nd_yav_path_a
			every_owned_location = {
				every_pop = {
					limit = { culture = culture:sindhi }
					change_pop_culture = culture:yavana_culture
				}
			}
		}
	}
	option = {
		name = nd_yav.1.b
		add_country_modifier = {
			modifier = nd_yav_dharma_of_the_yavana
			years = 30
			mode = add_and_extend
		}
		change_societal_value = {
			type = spiritualist_vs_humanist
			value = societal_value_move_to_right
		}
		change_culture = culture:yavana_culture
		custom_tooltip = nd_yav_culture_conversion_tt
		hidden_effect = {
			set_variable = nd_yav_formed
			set_variable = nd_yav_path_b
			every_owned_location = {
				every_pop = {
					limit = { culture = culture:sindhi }
					change_pop_culture = culture:yavana_culture
				}
			}
		}
	}
}
