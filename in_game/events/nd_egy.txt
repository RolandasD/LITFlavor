nd_egy.1 = {
	type = country_event
	title = nd_egy.1.t
	desc = nd_egy.1.d
	fire_only_once = yes

	option = {
		name = nd_egy.1.a
		add_country_modifier = {
			modifier = nd_sword_of_the_sultan
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_egy_formed
			set_variable = nd_egy_path_a
		}
	}
	option = {
		name = nd_egy.1.b
		add_country_modifier = {
			modifier = nd_garden_of_the_nile
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_egy_formed
			set_variable = nd_egy_path_b
		}
	}
}
