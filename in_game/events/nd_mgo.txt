namespace = nd_mgo

# Formation event — fires when Mongolia is formed
nd_mgo.1 = {
	type = country_event
	title = nd_mgo.1.t
	desc = nd_mgo.1.d
	fire_only_once = yes

	option = {
		name = nd_mgo.1.a
		add_country_modifier = {
			modifier = nd_mgo_chinggisid_restoration
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mgo_formed
			set_variable = nd_mgo_path_a
		}
	}
	option = {
		name = nd_mgo.1.b
		add_country_modifier = {
			modifier = nd_mgo_yellow_hat_patronage
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mgo_formed
			set_variable = nd_mgo_path_b
		}
	}
}
