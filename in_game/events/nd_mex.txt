namespace = nd_mex

# Formation event — The Birth of Mexico
nd_mex.1 = {
	type = country_event
	title = nd_mex.1.t
	desc = nd_mex.1.d
	fire_only_once = yes

	option = {
		name = nd_mex.1.a
		add_country_modifier = {
			modifier = nd_tres_garantias
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mex_formed
			set_variable = nd_mex_path_a
		}
	}
	option = {
		name = nd_mex.1.b
		add_country_modifier = {
			modifier = nd_grito_de_dolores
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mex_formed
			set_variable = nd_mex_path_b
		}
	}
}
