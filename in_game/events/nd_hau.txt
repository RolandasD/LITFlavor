namespace = nd_hau

# Formation event -- triggered from form_effect
nd_hau.1 = {
	type = country_event
	title = nd_hau.1.t
	desc = nd_hau.1.d
	fire_only_once = yes

	option = {
		name = nd_hau.1.a
		add_country_modifier = {
			modifier = nd_hau_walls_of_amina
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_hau_formed
			set_variable = nd_hau_path_a
		}
	}
	option = {
		name = nd_hau.1.b
		add_country_modifier = {
			modifier = nd_hau_crossroads_of_the_sahel
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_hau_formed
			set_variable = nd_hau_path_b
		}
	}
}
