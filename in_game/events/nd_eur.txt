namespace = nd_eur

# Formation event — fires when Europe is formed
nd_eur.1 = {
	type = country_event
	title = nd_eur.1.t
	desc = nd_eur.1.d
	fire_only_once = yes

	option = {
		name = nd_eur.1.a
		add_country_modifier = {
			modifier = nd_eur_universal_empire
			years = 60
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_large_move_to_left
		}
		hidden_effect = {
			set_variable = nd_eur_formed
			set_variable = nd_eur_path_a
		}
	}
	option = {
		name = nd_eur.1.b
		add_country_modifier = {
			modifier = nd_eur_concert_of_sovereigns
			years = 60
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_large_move_to_right
		}
		hidden_effect = {
			set_variable = nd_eur_formed
			set_variable = nd_eur_path_b
		}
	}
}
