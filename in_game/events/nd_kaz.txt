### Formation Event ###

nd_kaz.1 = {
	type = country_event
	title = nd_kaz.1.t
	desc = nd_kaz.1.d
	fire_only_once = yes

	option = {
		name = nd_kaz.1.a
		add_country_modifier = {
			modifier = nd_kaz_legacy_of_kasym
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_kaz_formed
			set_variable = nd_kaz_path_a
		}
	}
	option = {
		name = nd_kaz.1.b
		add_country_modifier = {
			modifier = nd_kaz_three_zhuz_confederacy
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_kaz_formed

		}
	}
}
