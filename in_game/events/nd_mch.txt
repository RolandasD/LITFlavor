# Formation Event — The Manchu Destiny
nd_mch.1 = {
	type = country_event
	title = nd_mch.1.t
	desc = nd_mch.1.d
	fire_only_once = yes

	option = {
		name = nd_mch.1.a
		add_country_modifier = {
			modifier = nd_mch_way_of_the_ancestors
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mch_formed
			set_variable = nd_mch_path_a
		}
	}
	option = {
		name = nd_mch.1.b
		add_country_modifier = {
			modifier = nd_mch_dragon_throne
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mch_formed
			set_variable = nd_mch_path_b
		}
	}
}
