namespace = nd_pol

# Formation event — fires from form_effect via trigger_event_non_silently
nd_pol.1 = {
	type = country_event
	title = nd_pol.1.t
	desc = nd_pol.1.d
	fire_only_once = yes

	option = {
		name = nd_pol.1.a
		add_country_modifier = {
			modifier = nd_pol_golden_liberty
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_pol_formed
			set_variable = nd_pol_path_a
		}
	}
	option = {
		name = nd_pol.1.b
		add_country_modifier = {
			modifier = nd_pol_piast_legacy
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_pol_formed
			set_variable = nd_pol_path_b
		}
	}
}
