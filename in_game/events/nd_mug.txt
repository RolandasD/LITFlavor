nd_mug.1 = {
	type = country_event
	title = nd_mug.1.t
	desc = nd_mug.1.d
	fire_only_once = yes

	option = {
		name = nd_mug.1.a
		add_country_modifier = {
			modifier = nd_sword_of_babur
			years = 40
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mug_formed
			set_variable = nd_mug_path_a
		}
	}
	option = {
		name = nd_mug.1.b
		add_country_modifier = {
			modifier = nd_light_of_akbar
			years = 40
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mug_formed
			set_variable = nd_mug_path_b
		}
	}
}
