namespace = nd_rom

nd_rom.1 = {
	type = country_event
	title = nd_rom.1.t
	desc = nd_rom.1.d
	fire_only_once = yes

	option = {
		name = nd_rom.1.a
		add_country_modifier = {
			modifier = nd_rom_imperium_restitutum
			years = 50
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_rom_formed
			set_variable = nd_rom_path_a
		}
	}
	option = {
		name = nd_rom.1.b
		add_country_modifier = {
			modifier = nd_rom_lux_aeterna
			years = 50
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_rom_formed
			set_variable = nd_rom_path_b
		}
	}
}
