# Formation Event — The Kingdom of Nepal
nd_npl.1 = {
	type = country_event
	title = nd_npl.1.t
	desc = nd_npl.1.d
	fire_only_once = yes

	option = {
		name = nd_npl.1.a
		add_country_modifier = {
			modifier = nd_gorkhali_conquest
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_npl_formed
			set_variable = nd_npl_path_a
		}
	}
	option = {
		name = nd_npl.1.b
		add_country_modifier = {
			modifier = nd_valley_of_temples
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_npl_formed
			set_variable = nd_npl_path_b
		}
	}
}
