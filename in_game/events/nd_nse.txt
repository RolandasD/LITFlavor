namespace = nd_nse

# Formation Event — The North Sea Empire is Reborn
nd_nse.1 = {
	type = country_event
	title = nd_nse.1.t
	desc = nd_nse.1.d
	fire_only_once = yes

	option = {
		name = nd_nse.1.a
		add_country_modifier = {
			modifier = nd_nse_whale_road
			years = 40
			mode = add_and_extend
		}
		change_societal_value = {
			type = land_vs_naval
			value = societal_value_large_move_to_right
		}
		hidden_effect = {
			set_variable = nd_nse_formed
			set_variable = nd_nse_path_a
		}
	}
	option = {
		name = nd_nse.1.b
		add_country_modifier = {
			modifier = nd_nse_crown_of_three_kingdoms
			years = 40
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_large_move_to_left
		}
		hidden_effect = {
			set_variable = nd_nse_formed
			set_variable = nd_nse_path_b
		}
	}
}
