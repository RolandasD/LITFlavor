# Formation event — The Founding of the Sokoto Caliphate
nd_sko.1 = {
	type = country_event
	title = nd_sko.1.t
	desc = nd_sko.1.d
	fire_only_once = yes

	option = {
		name = nd_sko.1.a
		add_country_modifier = {
			modifier = nd_light_of_knowledge
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_sko_formed
			set_variable = nd_sko_path_a
		}
	}
	option = {
		name = nd_sko.1.b
		add_country_modifier = {
			modifier = nd_banner_of_jihad
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_sko_formed

		}
	}
}
