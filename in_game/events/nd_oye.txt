namespace = nd_oye

# Formation Event — The Oyo Empire is proclaimed
nd_oye.1 = {
	type = country_event
	title = nd_oye.1.t
	desc = nd_oye.1.d
	fire_only_once = yes

	option = {
		name = nd_oye.1.a
		add_country_modifier = {
			modifier = nd_thunder_of_sango
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_oye_formed
			set_variable = nd_oye_path_a
		}
	}
	option = {
		name = nd_oye.1.b
		add_country_modifier = {
			modifier = nd_voice_of_the_oyo_mesi
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_oye_formed
			set_variable = nd_oye_path_b
		}
	}
}
