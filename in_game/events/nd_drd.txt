namespace = nd_drd

nd_drd.1 = {
	type = country_event
	title = nd_drd.1.t
	desc = nd_drd.1.d
	fire_only_once = yes

	option = {
		name = nd_drd.1.a
		add_country_modifier = {
			modifier = nd_sea_kings_dominion
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = land_vs_naval
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_drd_formed
			set_variable = nd_drd_path_a
		}
	}
	option = {
		name = nd_drd.1.b
		add_country_modifier = {
			modifier = nd_light_of_iona
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = spiritualist_vs_humanist
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_drd_formed
			set_variable = nd_drd_path_b
		}
	}
}
