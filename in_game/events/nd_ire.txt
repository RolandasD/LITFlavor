namespace = nd_ire

# Formation Event -- Ireland Unified
nd_ire.1 = {
	type = country_event
	title = nd_ire.1.t
	desc = nd_ire.1.d
	fire_only_once = yes

	option = {
		name = nd_ire.1.a
		add_country_modifier = {
			modifier = nd_ire_gaelic_resurgence
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = traditionalist_vs_innovative
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_ire_formed
			set_variable = nd_ire_path_a
		}
	}
	option = {
		name = nd_ire.1.b
		add_country_modifier = {
			modifier = nd_ire_atlantic_kingdom
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = outward_vs_inward
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_ire_formed
			set_variable = nd_ire_path_b
		}
	}
}
