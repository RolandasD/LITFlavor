# ============================================================
# Formation Event — The Founding of the Luba Empire
# ============================================================
nd_lbe.1 = {
	type = country_event
	title = nd_lbe.1.t
	desc = nd_lbe.1.d
	fire_only_once = yes

	option = {
		name = nd_lbe.1.a
		add_country_modifier = {
			modifier = nd_divine_authority_of_the_mulopwe
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_lbe_formed
			set_variable = nd_lbe_path_a
		}
	}
	option = {
		name = nd_lbe.1.b
		add_country_modifier = {
			modifier = nd_wisdom_of_the_bambudye
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_lbe_formed
			set_variable = nd_lbe_path_b
		}
	}
}
