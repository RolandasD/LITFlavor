namespace = nd_kbu

# Formation event — fires when Kanem-Bornu is formed
nd_kbu.1 = {
	type = country_event
	title = nd_kbu.1.t
	desc = nd_kbu.1.d
	fire_only_once = yes

	option = {
		name = nd_kbu.1.a
		add_country_modifier = {
			modifier = nd_sword_of_the_mai
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_kbu_formed
			set_variable = nd_kbu_path_a
		}
	}
	option = {
		name = nd_kbu.1.b
		add_country_modifier = {
			modifier = nd_crossroads_of_the_sahara
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_kbu_formed
			set_variable = nd_kbu_path_b
		}
	}
}
