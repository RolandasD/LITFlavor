nd_ilk.1 = {
	type = country_event
	title = nd_ilk.1.t
	desc = nd_ilk.1.d
	fire_only_once = yes

	option = {
		name = nd_ilk.1.a
		add_country_modifier = {
			modifier = nd_scourge_of_the_caliphate

		}

}
