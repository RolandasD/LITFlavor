# Formation Event
##############################

nd_mol.1 = {
	type = country_event
	title = nd_mol.1.t
	desc = nd_mol.1.d
	fire_only_once = yes

	option = {
		name = nd_mol.1.a
		add_country_modifier = {
			modifier = nd_mol_shield_of_christendom
			years = 15
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_mol_formed
			set_variable = nd_mol_path_a
		}
	}
	option = {
		name = nd_mol.1.b
		add_country_modifier = {
			modifier = nd_mol_crossroads_principality
			years = 15
			mode = add_and_extend
		}
		change_societal_value = {
			type = outward_vs_inward
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_mol_formed
			set_variable = nd_mol_path_b
		}
	}
}
