namespace = nd_ayu

# DHE Event 1: The Island Between Rivers
# Fires when the player completes nd_ayu_water_defenses
# Celebrates Ayutthaya's unique island fortress concept
nd_ayu.1 = {
	type = country_event
	fire_only_once = yes

	dynamic_historical_event = {
		tag = AYU
		from = 1337.1.1
		to = 1836.1.1
		monthly_chance = 100
	}

	trigger = {
		has_advance = nd_ayu_water_defenses
	}

	title = nd_ayu.1.t
	desc = nd_ayu.1.d

	option = {
		name = nd_ayu.1.a
		add_country_modifier = {
			modifier = nd_ayu_impregnable_island
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_ayu_formed
			set_variable = nd_ayu_path_a
		}
	}
	option = {
		name = nd_ayu.1.b
		add_country_modifier = {
			modifier = nd_ayu_river_crossroads
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_ayu_formed
			set_variable = nd_ayu_path_b
		}
	}
}
