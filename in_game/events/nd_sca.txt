namespace = nd_sca

# Formation Event — The Unification of Scandinavia
nd_sca.1 = {
	type = country_event
	title = nd_sca.1.t
	desc = nd_sca.1.d
	fire_only_once = yes

	option = {
		name = nd_sca.1.a
		add_country_modifier = {
			modifier = nd_sca_lords_of_the_northern_seas
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_sca_formed
			set_variable = nd_sca_path_a
		}
	}
	option = {
		name = nd_sca.1.b
		add_country_modifier = {
			modifier = nd_sca_all_thing_of_the_north
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_sca_formed
			set_variable = nd_sca_path_b
		}
	}
}
