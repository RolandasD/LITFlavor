nd_msi.1 = {
	type = country_event
	title = nd_msi.1.t
	desc = nd_msi.1.d
	fire_only_once = yes

	option = {
		name = nd_msi.1.a
		add_country_modifier = {
			modifier = nd_riders_of_the_naam
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_msi_formed
			set_variable = nd_msi_path_a
		}
	}
	option = {
		name = nd_msi.1.b
		add_country_modifier = {
			modifier = nd_mogho_naaba_court
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_msi_formed
			set_variable = nd_msi_path_b
		}
	}
}
