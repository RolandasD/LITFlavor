namespace = nd_rus

# Formation event - fires from form_effect
nd_rus.1 = {
	type = country_event
	title = nd_rus.1.t
	desc = nd_rus.1.d
	fire_only_once = yes

	option = {
		name = nd_rus.1.a
		add_country_modifier = {
			modifier = nd_rus_third_rome_ascendant
			years = 40
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_rus_formed
			set_variable = nd_rus_path_a
		}
	}
	option = {
		name = nd_rus.1.b
		add_country_modifier = {
			modifier = nd_rus_empire_of_the_steppe
			years = 40
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_rus_formed
			set_variable = nd_rus_path_b
		}
	}
}
