namespace = nd_iro

# Formation event — fires from form_effect
nd_iro.1 = {
	type = country_event
	title = nd_iro.1.t
	desc = nd_iro.1.d
	fire_only_once = yes

	option = {
		name = nd_iro.1.a
		add_country_modifier = {
			modifier = nd_great_law_of_peace
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_iro_formed
			set_variable = nd_iro_path_a
		}
	}
	option = {
		name = nd_iro.1.b
		add_country_modifier = {
			modifier = nd_war_road_of_the_longhouse
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_iro_formed
			set_variable = nd_iro_path_b
		}
	}
}
