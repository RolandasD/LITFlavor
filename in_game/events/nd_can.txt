namespace = nd_can

# Formation Event — The Birth of Canada
nd_can.1 = {
	type = country_event
	title = nd_can.1.t
	desc = nd_can.1.d
	fire_only_once = yes

	option = {
		name = nd_can.1.a
		add_country_modifier = {
			modifier = nd_northern_dominion
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_can_formed
			set_variable = nd_can_path_a
		}
	}
	option = {
		name = nd_can.1.b
		add_country_modifier = {
			modifier = nd_great_trading_nation
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_can_formed
			set_variable = nd_can_path_b
		}
	}
}
