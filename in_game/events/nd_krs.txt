nd_krs.1 = {
	type = country_event
	title = nd_krs.1.t
	desc = nd_krs.1.d
	fire_only_once = yes

	option = {
		name = nd_krs.1.a
		add_country_modifier = {
			modifier = nd_legacy_of_saladin
			years = 30
			mode = add_and_extend
		}

		}

	option = {
		name = nd_krs.1.b

		}

		}
	}
}
