# Advance DHE event 1 — The Undang-Undang Melaka
nd_mlc.1 = {
	type = country_event
	fire_only_once = yes

	dynamic_historical_event = {
		tag = MLC
		from = 1337.1.1
		to = 1836.1.1
		monthly_chance = 100
	}

	trigger = {
		has_advance = nd_mlc_undang_undang
	}

	title = nd_mlc.1.t
	desc = nd_mlc.1.d

	option = {
		name = nd_mlc.1.a
		add_country_modifier = {
			modifier = nd_mlc_law_of_the_sultanate
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mlc_formed
			set_variable = nd_mlc_path_a
		}
	}
	option = {
		name = nd_mlc.1.b
		add_country_modifier = {
			modifier = nd_mlc_maritime_law
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mlc_formed
			set_variable = nd_mlc_path_b
		}
	}
}
