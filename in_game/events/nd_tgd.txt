namespace = nd_tgd

# Formation event — The Tang Restoration
nd_tgd.1 = {
	type = country_event
	title = nd_tgd.1.t
	desc = nd_tgd.1.d
	fire_only_once = yes

	option = {
		name = nd_tgd.1.a
		add_country_modifier = {
			modifier = nd_tgd_restore_protectorates
			years = 40
			mode = add_and_extend
		}
		change_societal_value = {
			type = outward_vs_inward
			value = societal_value_large_move_to_left
		}
		change_societal_value = {
			type = belligerent_vs_conciliatory
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_tgd_formed
			set_variable = nd_tgd_path_a
		}
	}
	option = {
		name = nd_tgd.1.b
		add_country_modifier = {
			modifier = nd_tgd_golden_age_of_letters
			years = 40
			mode = add_and_extend
		}
		change_societal_value = {
			type = spiritualist_vs_humanist
			value = societal_value_large_move_to_right
		}
		change_societal_value = {
			type = traditionalist_vs_innovative
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_tgd_formed
			set_variable = nd_tgd_path_b
		}
	}
}
