##############################

nd_kon.1 = {
	type = country_event
	title = nd_kon.1.t
	desc = nd_kon.1.d
	fire_only_once = yes

	option = {
		name = nd_kon.1.a
		add_country_modifier = {
			modifier = nd_kon_christian_diplomacy

			mode = add_and_extend
		}

		}
	}

		name = nd_kon.1.b

			mode = add_and_extend
		}
