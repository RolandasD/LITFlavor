##############################
# Formation Event
##############################

nd_air.1 = {
	type = country_event
	title = nd_air.1.t
	desc = nd_air.1.d
	fire_only_once = yes

	option = {
		name = nd_air.1.a
		add_country_modifier = {
			modifier = nd_air_desert_sovereignty
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_air_formed
			set_variable = nd_air_path_a
		}
	}
	option = {
		name = nd_air.1.b
		add_country_modifier = {
			modifier = nd_air_gateway_of_the_sahara
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_air_formed
			set_variable = nd_air_path_b
		}
	}
}
