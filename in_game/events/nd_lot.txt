namespace = nd_lot

# Formation event — fired from form_effect via trigger_event_non_silently
nd_lot.1 = {
	type = country_event
	title = nd_lot.1.t
	desc = nd_lot.1.d
	fire_only_once = yes

	option = {
		name = nd_lot.1.a
		add_country_modifier = {
			modifier = nd_lot_crossroads_of_christendom
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_lot_formed
			set_variable = nd_lot_path_a
		}
	}
	option = {
		name = nd_lot.1.b
		add_country_modifier = {
			modifier = nd_lot_iron_crown_of_lothar
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_lot_formed
			set_variable = nd_lot_path_b
		}
	}
}
