namespace = nd_mor

# Formation Event — fires from form_effect
nd_mor.1 = {
	type = country_event
	title = nd_mor.1.t
	desc = nd_mor.1.d
	fire_only_once = yes

	option = {
		name = nd_mor.1.a
		add_country_modifier = {
			modifier = nd_sharif_commands_the_faithful
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mor_formed
			set_variable = nd_mor_path_a
		}
	}
	option = {
		name = nd_mor.1.b
		add_country_modifier = {
			modifier = nd_gateway_of_two_seas
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mor_formed
			set_variable = nd_mor_path_b
		}
	}
}
