namespace = nd_hab

# Formation event — fires when Austria is formed
nd_hab.1 = {
	type = country_event
	title = nd_hab.1.t
	desc = nd_hab.1.d
	fire_only_once = yes

	option = {
		name = nd_hab.1.a
		add_country_modifier = {
			modifier = nd_hab_aeiou_dynasty
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_hab_formed
			set_variable = nd_hab_path_a
		}
	}
	option = {
		name = nd_hab.1.b
		add_country_modifier = {
			modifier = nd_hab_antemurale
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_hab_formed
			set_variable = nd_hab_path_b
		}
	}
}
