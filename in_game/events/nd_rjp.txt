# Formation Event — The Rise of Rajputana
nd_rjp.1 = {
	type = country_event
	title = nd_rjp.1.t
	desc = nd_rjp.1.d
	fire_only_once = yes

	option = {
		name = nd_rjp.1.a
		add_country_modifier = {
			modifier = nd_rjp_sword_of_rajputana
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_rjp_formed
			set_variable = nd_rjp_path_a
		}
	}
	option = {
		name = nd_rjp.1.b
