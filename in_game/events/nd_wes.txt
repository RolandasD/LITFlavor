namespace = nd_wes

# Formation Event — The Rise of Westphalia
nd_wes.1 = {
	type = country_event
	title = nd_wes.1.t
	desc = nd_wes.1.d
	fire_only_once = yes

	option = {
		name = nd_wes.1.a
		add_country_modifier = {
			modifier = nd_westphalian_sovereignty
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = belligerent_vs_conciliatory
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_wes_formed
			set_variable = nd_wes_path_a
		}
	}
	option = {
		name = nd_wes.1.b
		add_country_modifier = {
			modifier = nd_red_earth_justice
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_wes_formed
			set_variable = nd_wes_path_b
		}
	}
}
