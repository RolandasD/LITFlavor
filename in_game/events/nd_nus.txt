# Formation Event — Nusantara
nd_nus.1 = {
	type = country_event
	title = nd_nus.1.t
	desc = nd_nus.1.d
	fire_only_once = yes

	option = {
		name = nd_nus.1.a
		add_country_modifier = {
			modifier = nd_nus_mandala_thalassocracy
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_nus_formed
			set_variable = nd_nus_path_a
		}
	}
	option = {
		name = nd_nus.1.b
		add_country_modifier = {
			modifier = nd_nus_spice_archipelago

			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_nus_formed

		}
	}
}
