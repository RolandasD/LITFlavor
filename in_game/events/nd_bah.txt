namespace = nd_bah

# Advance DHE event 1 — Mahmud Gawan's Reforms
nd_bah.1 = {
	type = country_event
	fire_only_once = yes

	dynamic_historical_event = {
		tag = BAH
		from = 1337.1.1
		to = 1836.1.1
		monthly_chance = 100
	}

	trigger = {
		has_advance = nd_bah_gawan_reforms
	}

	title = nd_bah.1.t
	desc = nd_bah.1.d

	option = {
		name = nd_bah.1.a
		add_country_modifier = {
			modifier = nd_bah_gawan_centralization
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_bah_formed
			set_variable = nd_bah_path_a
		}
	}
	option = {
		name = nd_bah.1.b
		add_country_modifier = {
			modifier = nd_bah_gawan_tolerance
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_bah_formed
			set_variable = nd_bah_path_b
		}
	}
}
