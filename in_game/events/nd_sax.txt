namespace = nd_sax

# Formation event — fires from form_effect
nd_sax.1 = {
	type = country_event
	title = nd_sax.1.t
	desc = nd_sax.1.d
	fire_only_once = yes

	option = {
		name = nd_sax.1.a
		add_country_modifier = {
			modifier = nd_wittenberg_legacy
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = traditionalist_vs_innovative
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_sax_formed
			set_variable = nd_sax_path_a
		}
	}
	option = {
		name = nd_sax.1.b
		add_country_modifier = {
			modifier = nd_augustan_splendor
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = aristocracy_vs_plutocracy
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_sax_formed
			set_variable = nd_sax_path_b
		}
	}
}
