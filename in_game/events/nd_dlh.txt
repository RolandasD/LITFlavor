namespace = nd_dlh

# Formation event — fires from form_effect
nd_dlh.1 = {
	type = country_event
	title = nd_dlh.1.t
	desc = nd_dlh.1.d
	fire_only_once = yes

	option = {
		name = nd_dlh.1.a
		add_country_modifier = {
			modifier = nd_sword_of_the_sultanate
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_dlh_formed
			set_variable = nd_dlh_path_a
		}
	}
	option = {
		name = nd_dlh.1.b
		add_country_modifier = {
			modifier = nd_throne_of_hindustan
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_dlh_formed
			set_variable = nd_dlh_path_b
		}
	}
}
