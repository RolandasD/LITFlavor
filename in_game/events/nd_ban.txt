namespace = nd_ban

nd_ban.1 = {
	type = country_event
	title = nd_ban.1.t
	desc = nd_ban.1.d
	fire_only_once = yes

	option = {
		name = nd_ban.1.a
		add_country_modifier = {
			modifier = nd_pepper_emporium
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_ban_formed
			set_variable = nd_ban_path_a
		}
	}
	option = {
		name = nd_ban.1.b
		add_country_modifier = {
			modifier = nd_sunda_strait_guardians
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_ban_formed
			set_variable = nd_ban_path_b
		}
	}
}
