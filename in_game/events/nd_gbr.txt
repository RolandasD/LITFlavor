namespace = nd_gbr

# Formation Event — Act of Union
nd_gbr.1 = {
	type = country_event
	title = nd_gbr.1.t
	desc = nd_gbr.1.d
	fire_only_once = yes

	option = {
		name = nd_gbr.1.a
		add_country_modifier = {
			modifier = nd_rule_britannia
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_gbr_formed
			set_variable = nd_gbr_path_a
		}
	}
	option = {
		name = nd_gbr.1.b
		add_country_modifier = {
			modifier = nd_workshop_of_the_world
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_gbr_formed
			set_variable = nd_gbr_path_b
		}
	}
}
