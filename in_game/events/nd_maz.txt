### Formation Event ###
nd_maz.1 = {
	type = country_event
	title = nd_maz.1.t
	desc = nd_maz.1.d
	fire_only_once = yes

	option = {
		name = nd_maz.1.a
		add_country_modifier = {
			modifier = nd_maz_piast_legacy
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_maz_formed
			set_variable = nd_maz_path_a
		}
	}
	option = {
		name = nd_maz.1.b
		add_country_modifier = {
			modifier = nd_maz_gate_of_the_vistula
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_maz_formed
			set_variable = nd_maz_path_b
		}
	}
}
