namespace = nd_hen

# Formation Event - Hen Ogledd
nd_hen.1 = {
	type = country_event
	title = nd_hen.1.t
	desc = nd_hen.1.d
	fire_only_once = yes

	option = {
		name = nd_hen.1.a
		add_country_modifier = {
			modifier = nd_hen_yr_awen
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = traditionalist_vs_innovative
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_hen_formed
			set_variable = nd_hen_path_a
		}
	}
	option = {
		name = nd_hen.1.b
		add_country_modifier = {
			modifier = nd_hen_cleddyf_y_gogledd
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = belligerent_vs_conciliatory
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_hen_formed
			set_variable = nd_hen_path_b
		}
	}
}
