namespace = nd_cth

# Formation Event — The Rebirth of Carthage
nd_cth.1 = {
	type = country_event
	fire_only_once = yes

	title = nd_cth.1.t
	desc = nd_cth.1.d

	option = {
		name = nd_cth.1.a
		add_country_modifier = {
			modifier = nd_cth_thalassocracy
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = land_vs_naval
			value = societal_value_large_move_to_right
		}
		change_societal_value = {
			type = mercantilism_vs_free_trade
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_cth_formed
			set_variable = nd_cth_path_a
		}
	}

	option = {
		name = nd_cth.1.b
		add_country_modifier = {
			modifier = nd_cth_punic_imperium
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = belligerent_vs_conciliatory
			value = societal_value_large_move_to_left
		}
		change_societal_value = {
			type = quality_vs_quantity
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_cth_formed
			set_variable = nd_cth_path_b
		}
	}
}
