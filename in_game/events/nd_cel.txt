namespace = nd_cel

# Formation Event - Celtic Union
nd_cel.1 = {
	type = country_event
	title = nd_cel.1.t
	desc = nd_cel.1.d
	fire_only_once = yes

	option = {
		name = nd_cel.1.a
		add_country_modifier = {
			modifier = nd_cel_tir_na_saor
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_cel_formed
			set_variable = nd_cel_path_a
		}
	}
	option = {
		name = nd_cel.1.b
		add_country_modifier = {
			modifier = nd_cel_claidheamh_mor
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_cel_formed
			set_variable = nd_cel_path_b
		}
	}
}
