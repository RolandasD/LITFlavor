# Formation Event — The Restoration of Al-Andalus
nd_adu.1 = {
	type = country_event
	title = nd_adu.1.t
	desc = nd_adu.1.d
	fire_only_once = yes

	option = {
		name = nd_adu.1.a
		add_country_modifier = {
			modifier = nd_adu_ornament_of_the_world
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_adu_formed
			set_variable = nd_adu_path_a
		}
	}
	option = {
		name = nd_adu.1.b
		add_country_modifier = {

		}

	}
}
