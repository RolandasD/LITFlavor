namespace = nd_cso

# Formation Event — Song Dynasty Restoration
nd_cso.1 = {
	type = country_event
	title = nd_cso.1.t
	desc = nd_cso.1.d
	fire_only_once = yes

	option = {
		name = nd_cso.1.a
		add_country_modifier = {
			modifier = nd_cso_maritime_supremacy
			years = 30
			mode = add_and_extend
		}
		change_societal_value = {
			type = land_vs_naval
			value = societal_value_large_move_to_right
		}
		change_societal_value = {
			type = aristocracy_vs_plutocracy
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_cso_formed
			set_variable = nd_cso_path_a
		}
	}
	option = {
		name = nd_cso.1.b
		add_country_modifier = {
			modifier = nd_cso_confucian_meritocracy
			years = 30
			mode = add_and_extend
		}
		change_societal_value = {
			type = traditionalist_vs_innovative
			value = societal_value_large_move_to_right
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_cso_formed
			set_variable = nd_cso_path_b
		}
	}
}
