namespace = nd_ukr

# Formation event — Ruthenia rises
nd_ukr.1 = {
	type = country_event
	title = nd_ukr.1.t
	desc = nd_ukr.1.d
	fire_only_once = yes

	option = {
		name = nd_ukr.1.a
		add_country_modifier = {
			modifier = nd_ukr_sich_republic
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_ukr_formed
			set_variable = nd_ukr_path_a
		}
	}
	option = {
		name = nd_ukr.1.b
		add_country_modifier = {
			modifier = nd_ukr_breadbasket
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_ukr_formed
			set_variable = nd_ukr_path_b
		}
	}
}
