namespace = nd_lva

# Formation Event — The Birth of the Baltic Nation
nd_lva.1 = {
	type = country_event
	title = nd_lva.1.t
	desc = nd_lva.1.d
	fire_only_once = yes

	option = {
		name = nd_lva.1.a
		add_country_modifier = {
			modifier = nd_guardians_of_the_sacred_groves
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = traditionalist_vs_innovative
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_lva_formed
			set_variable = nd_lva_path_a
		}
	}
	option = {
		name = nd_lva.1.b
		add_country_modifier = {
			modifier = nd_lords_of_the_amber_coast
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = mercantilism_vs_free_trade
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_lva_formed
			set_variable = nd_lva_path_b
		}
	}
}
