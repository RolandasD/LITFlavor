# Formation Event — The United Pueblo
nd_pue.1 = {
	type = country_event
	title = nd_pue.1.t
	desc = nd_pue.1.d
	fire_only_once = yes

	option = {
		name = nd_pue.1.a
		add_country_modifier = {
			modifier = nd_path_of_pope
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_pue_formed
			set_variable = nd_pue_path_a
		}
	}
	option = {
		name = nd_pue.1.b
		add_country_modifier = {
			modifier = nd_path_of_the_kiva
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_pue_formed
			set_variable = nd_pue_path_b
		}
	}
}
