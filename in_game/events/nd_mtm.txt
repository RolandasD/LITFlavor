namespace = nd_mtm

# nd_mtm.1 — Sultan Agung's Conquests (Age 3)
nd_mtm.1 = {
	type = country_event
	fire_only_once = yes

	dynamic_historical_event = {
		tag = MTM
		from = 1337.1.1
		to = 1836.1.1
		monthly_chance = 100
	}

	trigger = {
		has_advance = nd_mtm_agung_conquests
	}

	title = nd_mtm.1.t
	desc = nd_mtm.1.d

	option = {
		name = nd_mtm.1.a
		add_country_modifier = {
			modifier = nd_mtm_agung_military_legacy
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mtm_formed
			set_variable = nd_mtm_path_a
		}
	}
	option = {
		name = nd_mtm.1.b
		add_country_modifier = {
			modifier = nd_mtm_agung_administrative_legacy
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mtm_formed
			set_variable = nd_mtm_path_b
		}
	}
}
