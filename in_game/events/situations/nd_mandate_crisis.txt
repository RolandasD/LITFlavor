# Cultural Rebellion (dynamic culture warlord)
nd_mandate_crisis.21 = {
	type = country_event
	category = situation_event
	image = "gfx/interface/illustrations/disaster/crisis_of_the_chinese_dynasty.dds"

	title = nd_mandate_crisis.21.title
	desc = nd_mandate_crisis.21.desc

	trigger = {
		any_owned_location = {
			NOT = { dominant_culture = root.culture }
			province != root.capital.province
		}
	}

	immediate = {
		ruler_or_regent = { save_scope_as = target_character }
		# Find the largest non-primary culture
		ordered_owned_location = {
			limit = {
				NOT = { dominant_culture = root.culture }
				province != root.capital.province
			}
			order_by = {
				value = population
				if = {
					limit = { location_rank = location_rank:city }
					multiply = 10
				}
				if = {
					limit = { location_rank = location_rank:town }
					multiply = 5
				}
			}
			max = 1
			check_range_bounds = no
			dominant_culture = { save_scope_as = target_culture }
			save_scope_as = target_location
		}
	}

	option = {
		name = nd_mandate_crisis.21.a

		if = {
			limit = { exists = scope:target_location }
			scope:target_location = {
				create_dynasty_from_location = random
				last_dynasty_in_location = { save_scope_as = target_dynasty }
				create_country_from_location = {
					save_scope_as = target_country
					nd_mc_released_country_setup = yes
					set_capital = prev
					create_character = {
						culture = scope:target_culture
						age = 30
						dynasty = scope:target_dynasty
						save_scope_as = new_ruler
						prev = { set_new_ruler = prev }
					}
				}
			}
			if = {
				limit = { exists = scope:target_country }
				# Transfer locations of that culture
				every_owned_location = {
					limit = {
						dominant_culture = scope:target_culture
						province != root.capital.province
					}
					change_location_owner = scope:target_country
				}
				nd_mc_add_country_to_challengers_and_dow = { country = scope:target_country }
			}
		}

		# Fire warlord flavor event to the released country
		hidden_effect = {
			if = {
				limit = { exists = scope:target_country }
				scope:target_country = {
					trigger_event_silently = { id = nd_mandate_crisis.30 days = { 20 60 } }
				}
			}
			change_variable = { name = nd_mc_tags_released add = 1 }
		}
	}

	option = {
		name = nd_mandate_crisis.21.play_as_tag
		trigger = { hidden_trigger = { is_ai = no } }

		if = {
			limit = { exists = scope:target_location }
			scope:target_location = {
				create_dynasty_from_location = random
				last_dynasty_in_location = { save_scope_as = target_dynasty }
				create_country_from_location = {
					save_scope_as = target_country
					nd_mc_released_country_setup = yes
					set_capital = prev
					create_character = {
						culture = scope:target_culture
						age = 30
						dynasty = scope:target_dynasty
						save_scope_as = new_ruler
						prev = { set_new_ruler = prev }
					}
				}
			}
			if = {
				limit = { exists = scope:target_country }
				every_owned_location = {
					limit = {
						dominant_culture = scope:target_culture
						province != root.capital.province
					}
					change_location_owner = scope:target_country
				}
				nd_mc_add_country_to_challengers_and_dow = { country = scope:target_country }
				change_player = scope:target_country
			}
		}

		hidden_effect = {
			change_variable = { name = nd_mc_tags_released add = 1 }
		}
	}
}
