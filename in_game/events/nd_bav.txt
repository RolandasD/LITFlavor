namespace = nd_bav

# Formation event — fires from form_effect
nd_bav.1 = {
	type = country_event
	title = nd_bav.1.t
	desc = nd_bav.1.d
	fire_only_once = yes

	option = {
		name = nd_bav.1.a
		add_country_modifier = {
			modifier = nd_catholic_shield
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_bav_formed
			set_variable = nd_bav_path_a
		}
	}
	option = {
		name = nd_bav.1.b
		add_country_modifier = {
			modifier = nd_wittelsbach_inheritance
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_bav_formed
			set_variable = nd_bav_path_b
		}
	}
}
