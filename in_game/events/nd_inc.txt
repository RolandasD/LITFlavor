nd_inc.1 = {
	type = country_event
	title = nd_inc.1.t
	desc = nd_inc.1.d
	fire_only_once = yes

	option = {
		name = nd_inc.1.a
		add_country_modifier = {
			modifier = nd_children_of_inti
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_inc_formed
			set_variable = nd_inc_path_a
		}
	}
	option = {
		name = nd_inc.1.b
		add_country_modifier = {
			modifier = nd_lords_of_the_four_quarters
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_inc_formed
			set_variable = nd_inc_path_b
		}
	}
}
