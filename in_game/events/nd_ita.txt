namespace = nd_ita

# Formation event — fires from form_effect
nd_ita.1 = {
	type = country_event
	title = nd_ita.1.t
	desc = nd_ita.1.d
	fire_only_once = yes

	option = {
		name = nd_ita.1.a
		add_country_modifier = {
			modifier = nd_ita_patria_unita
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_ita_formed
			set_variable = nd_ita_path_a
		}
	}
	option = {
		name = nd_ita.1.b
		add_country_modifier = {
			modifier = nd_ita_repubblica_delle_citta
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_ita_formed
			set_variable = nd_ita_path_b
		}
	}
}
