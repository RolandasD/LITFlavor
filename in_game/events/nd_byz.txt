namespace = nd_byz

# Formation event — fires from form_effect
nd_byz.1 = {
	type = country_event
	title = nd_byz.1.t
	desc = nd_byz.1.d
	fire_only_once = yes

	option = {
		name = nd_byz.1.a
		add_country_modifier = {
			modifier = nd_byz_renovatio_imperii
			years = 40
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_large_move_to_left
		}
		hidden_effect = {
			set_variable = nd_byz_formed
			set_variable = nd_byz_path_a
		}
	}
	option = {
		name = nd_byz.1.b
		add_country_modifier = {
			modifier = nd_byz_orthodox_ecumene
			years = 40
			mode = add_and_extend
		}
		change_societal_value = {
			type = spiritualist_vs_humanist
			value = societal_value_large_move_to_left
		}
		hidden_effect = {
			set_variable = nd_byz_formed
			set_variable = nd_byz_path_b
		}
	}
}
