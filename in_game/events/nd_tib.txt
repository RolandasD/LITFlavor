namespace = nd_tib

# Formation event - fires from form_effect
nd_tib.1 = {
	type = country_event
	title = nd_tib.1.t
	desc = nd_tib.1.d
	fire_only_once = yes

	option = {
		name = nd_tib.1.a
		add_country_modifier = {
			modifier = nd_tib_ganden_phodrang
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_tib_formed
			set_variable = nd_tib_path_a
		}
	}
	option = {
		name = nd_tib.1.b
		add_country_modifier = {
			modifier = nd_tib_dharma_kings
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_tib_formed
			set_variable = nd_tib_path_b
		}
	}
}
