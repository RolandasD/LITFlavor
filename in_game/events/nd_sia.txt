namespace = nd_sia

# Formation event — triggered by form_effect
nd_sia.1 = {
	type = country_event
	title = nd_sia.1.t
	desc = nd_sia.1.d
	fire_only_once = yes

	option = {
		name = nd_sia.1.a
		add_country_modifier = {
			modifier = nd_sia_warrior_kings
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_sia_formed
			set_variable = nd_sia_path_a
		}
	}
	option = {
		name = nd_sia.1.b
		add_country_modifier = {
			modifier = nd_sia_venice_of_the_east
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_sia_formed
			set_variable = nd_sia_path_b
		}
	}
}
