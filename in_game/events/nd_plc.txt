namespace = nd_plc

# Formation Event — The Union of Two Nations
nd_plc.1 = {
	type = country_event
	title = nd_plc.1.t
	desc = nd_plc.1.d
	fire_only_once = yes

	option = {
		name = nd_plc.1.a
		add_country_modifier = {
			modifier = nd_golden_liberty
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_plc_formed
			set_variable = nd_plc_path_a
		}
	}
	option = {
		name = nd_plc.1.b
		add_country_modifier = {
			modifier = nd_jagiellonian_union
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_plc_formed
			set_variable = nd_plc_path_b
		}
	}
}
