nd_azt.1 = {
	type = country_event
	title = nd_azt.1.t
	desc = nd_azt.1.d
	fire_only_once = yes

	option = {
		name = nd_azt.1.a
		add_country_modifier = {
			modifier = nd_way_of_flower_wars
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_azt_formed
			set_variable = nd_azt_path_a
		}
	}
	option = {
		name = nd_azt.1.b
		add_country_modifier = {
			modifier = nd_pochteca_road
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_azt_formed
			set_variable = nd_azt_path_b
		}
	}
}
