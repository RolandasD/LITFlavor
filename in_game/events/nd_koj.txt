nd_koj.1 = {
	type = country_event
	title = nd_koj.1.t
	desc = nd_koj.1.d
	fire_only_once = yes

	option = {
		name = nd_koj.1.a
		add_country_modifier = {
			modifier = nd_defender_of_the_faith
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_koj_formed
			set_variable = nd_koj_path_a
		}
	}
	option = {
		name = nd_koj.1.b
		add_country_modifier = {
			modifier = nd_crossroads_of_commerce
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_koj_formed
			set_variable = nd_koj_path_b
		}
	}
}
