namespace = nd_soi

# Formation event - fires from form_effect
nd_soi.1 = {
	type = country_event
	title = nd_soi.1.t
	desc = nd_soi.1.d
	fire_only_once = yes

	option = {
		name = nd_soi.1.a
		add_country_modifier = {
			modifier = nd_pearl_of_the_indian_ocean
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_soi_formed
			set_variable = nd_soi_path_a
		}
	}
	option = {
		name = nd_soi.1.b
		add_country_modifier = {
			modifier = nd_lords_of_the_waters
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_soi_formed
			set_variable = nd_soi_path_b
		}
	}
}
