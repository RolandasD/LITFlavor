namespace = nd_bel

# Formation event — fired from form_effect
nd_bel.1 = {
	type = country_event
	title = nd_bel.1.t
	desc = nd_bel.1.d
	fire_only_once = yes

	option = {
		name = nd_bel.1.a
		add_country_modifier = {
			modifier = nd_bel_workshop_of_the_world
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_bel_formed
			set_variable = nd_bel_path_a
		}
	}
	option = {
		name = nd_bel.1.b
		add_country_modifier = {
			modifier = nd_bel_scrap_of_paper
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_bel_formed
			set_variable = nd_bel_path_b
		}
	}
}
