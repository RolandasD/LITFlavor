# LITFlavor - Lithuanian Flavor Events
# Extends flavor for the Grand Duchy of Lithuania in EU5.
# Covers historical events from the medieval period through the early modern era.

namespace = lit_flavor

######################################
# Event 1: The Legacy of Mindaugas
# Historical context: Mindaugas was the only King of Lithuania,
# baptized in 1251, but later renounced Christianity.
######################################
event = {
    id = lit_flavor.1
    title = lit_flavor.1.t
    desc = lit_flavor.1.d
    picture = "gfx/event_pictures/eu5_throne_room.dds"

    type = country_event
    hidden = no
    fire_only_once = yes

    trigger = {
        tag = LIT
        NOT = { has_country_flag = lit_flavor_mindaugas_fired }
    }

    weight_multiplier = {
        base = 1
    }

    option = {
        name = lit_flavor.1.a
        # Embrace Christian heritage to boost relations with Catholic neighbors
        add_country_modifier = {
            name = lit_mindaugas_christian_legacy
            duration = 3650
        }
        set_country_flag = lit_flavor_mindaugas_fired
    }
    option = {
        name = lit_flavor.1.b
        # Reaffirm Baltic pagan traditions to boost national identity
        add_country_modifier = {
            name = lit_mindaugas_pagan_legacy
            duration = 3650
        }
        set_country_flag = lit_flavor_mindaugas_fired
    }
}
