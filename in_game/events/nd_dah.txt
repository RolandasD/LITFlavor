# No formation event — DAH is event-only (flavor_dah.1)
# Advance DHE events only

# nd_dah.1 — Ghezo's Military Reforms (triggered by nd_dah_ghezo_reforms advance)
nd_dah.1 = {
	type = country_event
	fire_only_once = yes

	dynamic_historical_event = {
		tag = DAH
		from = 1337.1.1
		to = 1836.1.1
		monthly_chance = 100
	}

	trigger = {
		has_advance = nd_dah_ghezo_reforms
	}

	title = nd_dah.1.t
	desc = nd_dah.1.d

	option = {
		name = nd_dah.1.a
		add_country_modifier = {
			modifier = nd_dah_standing_army
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_dah_formed
			set_variable = nd_dah_path_a
		}
	}
	option = {
		name = nd_dah.1.b
		add_country_modifier = {
			modifier = nd_dah_royal_monopoly
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_dah_formed
			set_variable = nd_dah_path_b
		}
	}
}
