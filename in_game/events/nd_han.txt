namespace = nd_han

# Formation event — fires from form_effect
nd_han.1 = {
	type = country_event
	title = nd_han.1.t
	desc = nd_han.1.d
	fire_only_once = yes

	option = {
		name = nd_han.1.a
		add_country_modifier = {
			modifier = nd_hanoverian_succession
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = outward_vs_inward
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_han_formed
			set_variable = nd_han_path_a
		}
	}
	option = {
		name = nd_han.1.b
		add_country_modifier = {
			modifier = nd_welf_renaissance
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = traditionalist_vs_innovative
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_han_formed
			set_variable = nd_han_path_b
		}
	}
}
