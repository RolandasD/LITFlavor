### Formation Event ###
nd_sil.1 = {
	type = country_event
	title = nd_sil.1.t
	desc = nd_sil.1.d
	fire_only_once = yes

	option = {
		name = nd_sil.1.a
		add_country_modifier = {
			modifier = nd_sil_crossroads_of_crowns
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_sil_formed
			set_variable = nd_sil_path_a
		}
	}
	option = {
		name = nd_sil.1.b
		add_country_modifier = {
			modifier = nd_sil_linen_and_silver
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_sil_formed

		}
	}
}
