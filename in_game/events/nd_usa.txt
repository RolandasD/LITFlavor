namespace = nd_usa

# Formation Event — Birth of the Republic
nd_usa.1 = {
	type = country_event
	title = nd_usa.1.t
	desc = nd_usa.1.d
	fire_only_once = yes

	option = {
		name = nd_usa.1.a
		add_country_modifier = {
			modifier = nd_usa_revolutionary_republic
			years = 40
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_usa_formed
			set_variable = nd_usa_path_a
		}
	}
	option = {
		name = nd_usa.1.b
		add_country_modifier = {
			modifier = nd_usa_manifest_destiny
			years = 40
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_usa_formed
			set_variable = nd_usa_path_b
		}
	}
	option = {
		name = nd_usa.1.c
		add_country_modifier = {
			modifier = nd_usa_city_on_a_hill
			years = 40
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_usa_formed
			set_variable = nd_usa_path_c
		}
	}
}
