namespace = nd_dal

##############################
# Formation Event
##############################

nd_dal.1 = {
	type = country_event
	title = nd_dal.1.t
	desc = nd_dal.1.d
	fire_only_once = yes

	option = {
		name = nd_dal.1.a
		add_country_modifier = {
			modifier = nd_dal_republic_of_the_adriatic
			years = 15
			mode = add_and_extend
		}
		change_societal_value = {
			type = outward_vs_inward
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_dal_formed
			set_variable = nd_dal_path_a
		}
	}
	option = {
		name = nd_dal.1.b
		add_country_modifier = {
			modifier = nd_dal_maritime_league
			years = 15
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_dal_formed
			set_variable = nd_dal_path_b
		}
	}
}
