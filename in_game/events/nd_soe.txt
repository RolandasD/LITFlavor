# Formation event - The Rise of Songhai
nd_soe.1 = {
	type = country_event
	title = nd_soe.1.t
	desc = nd_soe.1.d
	fire_only_once = yes

	option = {
		name = nd_soe.1.a
		add_country_modifier = {
			modifier = nd_soe_river_conqueror

			mode = add_and_extend
