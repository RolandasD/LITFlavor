# Formation event — The Birth of the Hellenic State
nd_gre.1 = {
	type = country_event
	title = nd_gre.1.t
	desc = nd_gre.1.d
	fire_only_once = yes

	option = {
		name = nd_gre.1.a
		add_country_modifier = {
			modifier = nd_gre_megali_idea
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_gre_formed
			set_variable = nd_gre_path_a
		}
	}
	option = {
		name = nd_gre.1.b
		add_country_modifier = {
			modifier = nd_gre_hellenic_enlightenment
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_gre_formed
			set_variable = nd_gre_path_b
		}
	}
}
