namespace = nd_ryu

# Formation event — fired from form_effect
nd_ryu.1 = {
	type = country_event
	title = nd_ryu.1.t
	desc = nd_ryu.1.d
	fire_only_once = yes

	option = {
		name = nd_ryu.1.a
		add_country_modifier = {
			modifier = nd_ryu_bridge_of_nations
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_ryu_formed
			set_variable = nd_ryu_path_a
		}
	}
	option = {
		name = nd_ryu.1.b
		add_country_modifier = {
			modifier = nd_ryu_sovereignty_of_isles
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_ryu_formed
			set_variable = nd_ryu_path_b
		}
	}
}
