# DHE Event 1: Adat basandi Syarak — the formal synthesis of matrilineal adat with Islamic law
nd_pgr.1 = {
	type = country_event
	fire_only_once = yes

	dynamic_historical_event = {
		tag = PGR
		from = 1337.1.1
		to = 1836.1.1
		monthly_chance = 100
	}

	trigger = {
		has_advance = nd_pgr_adat_basandi_syarak
	}

	title = nd_pgr.1.t
	desc = nd_pgr.1.d

	option = {
		name = nd_pgr.1.a
		add_country_modifier = {
			modifier = nd_pgr_syarak_harmony
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_pgr_formed
			set_variable = nd_pgr_path_a
		}
	}
	option = {
		name = nd_pgr.1.b
		add_country_modifier = {
			modifier = nd_pgr_adat_primacy
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_pgr_formed
			set_variable = nd_pgr_path_b
		}
	}
}
