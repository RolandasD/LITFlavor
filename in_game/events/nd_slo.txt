# Formation Event
##############################

nd_slo.1 = {
	type = country_event
	title = nd_slo.1.t
	desc = nd_slo.1.d
	fire_only_once = yes

	option = {
		name = nd_slo.1.a
		add_country_modifier = {
			modifier = nd_slo_mining_crown
			years = 30
			mode = add_and_extend
		}
		change_societal_value = {
			type = outward_vs_inward
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_slo_formed
			set_variable = nd_slo_path_a
		}
	}
	option = {
		name = nd_slo.1.b
		add_country_modifier = {
			modifier = nd_slo_frontier_bastion
			years = 30
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_slo_formed
			set_variable = nd_slo_path_b
		}
	}
}
