namespace = nd_kry

# Formation event — fired from form_effect
nd_kry.1 = {
	type = country_event
	title = nd_kry.1.t
	desc = nd_kry.1.d
	fire_only_once = yes

	option = {
		name = nd_kry.1.a
		add_country_modifier = {
			modifier = nd_kry_mountain_wall
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_kry_formed
			set_variable = nd_kry_path_a
		}
	}
	option = {
		name = nd_kry.1.b
		add_country_modifier = {
			modifier = nd_kry_caspian_silk_roads
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_kry_formed
			set_variable = nd_kry_path_b
		}
	}
}
