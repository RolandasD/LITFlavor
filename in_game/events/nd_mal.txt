namespace = nd_mal

# Formation event — The Restoration of the Manden Empire
nd_mal.1 = {
	type = country_event
	title = nd_mal.1.t
	desc = nd_mal.1.d
	fire_only_once = yes

	option = {
		name = nd_mal.1.a
		add_country_modifier = {
			modifier = nd_golden_throne_of_niani
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mal_formed
			set_variable = nd_mal_path_a
		}
	}
	option = {
		name = nd_mal.1.b
		add_country_modifier = {
			modifier = nd_mandate_of_the_gbara
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mal_formed
			set_variable = nd_mal_path_b
		}
	}
}
