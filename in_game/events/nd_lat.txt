namespace = nd_lat

# Formation Event - The Latin Empire
nd_lat.1 = {
	type = country_event
	title = nd_lat.1.t
	desc = nd_lat.1.d
	fire_only_once = yes

	option = {
		name = nd_lat.1.a
		add_country_modifier = {
			modifier = nd_lat_frankish_feudal_order
			years = 30
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_lat_formed
			set_variable = nd_lat_path_a
		}
	}
	option = {
		name = nd_lat.1.b
		add_country_modifier = {
			modifier = nd_lat_venetian_partnership
			years = 30
			mode = add_and_extend
		}
		change_societal_value = {
			type = aristocracy_vs_plutocracy
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_lat_formed
			set_variable = nd_lat_path_b
		}
	}
}
