# Formation event — The Birth of Malaya
nd_msa.1 = {
	type = country_event
	title = nd_msa.1.t
	desc = nd_msa.1.d
	fire_only_once = yes

	option = {
		name = nd_msa.1.a
		add_country_modifier = {
			modifier = nd_msa_thalassocratic_emporium
			years = 30
			mode = add_and_extend
		}

			set_variable = nd_msa_formed

		}
	}

		name = nd_msa.1.b
