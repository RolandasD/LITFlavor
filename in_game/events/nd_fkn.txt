namespace = nd_fkn

# Formation event — fires when player forms Franconia
nd_fkn.1 = {
	type = country_event
	title = nd_fkn.1.t
	desc = nd_fkn.1.d
	fire_only_once = yes

	option = {
		name = nd_fkn.1.a
		add_country_modifier = {
			modifier = nd_fkn_princes_faith
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = spiritualist_vs_humanist
			value = societal_value_minor_move_to_left
		}
		hidden_effect = {
			set_variable = nd_fkn_formed
			set_variable = nd_fkn_path_a
		}
	}
	option = {
		name = nd_fkn.1.b
		add_country_modifier = {
			modifier = nd_fkn_craftsmens_spirit
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = traditionalist_vs_innovative
			value = societal_value_minor_move_to_right
		}
		hidden_effect = {
			set_variable = nd_fkn_formed
			set_variable = nd_fkn_path_b
		}
	}
}
