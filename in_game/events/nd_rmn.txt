namespace = nd_rmn

# Formation event — fires from form_effect
nd_rmn.1 = {
	type = country_event
	title = nd_rmn.1.t
	desc = nd_rmn.1.d
	fire_only_once = yes

	option = {
		name = nd_rmn.1.a
		add_country_modifier = {
			modifier = nd_rmn_antemurale_christianitatis
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_rmn_formed
			set_variable = nd_rmn_path_a
		}
	}
	option = {
		name = nd_rmn.1.b
		add_country_modifier = {
			modifier = nd_rmn_danubian_prosperity
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_rmn_formed
			set_variable = nd_rmn_path_b
		}
	}
}
