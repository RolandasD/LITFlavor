namespace = nd_mca

nd_mca.1 = {
	type = country_event
	title = nd_mca.1.t
	desc = nd_mca.1.d
	fire_only_once = yes

	option = {
		name = nd_mca.1.a
		add_country_modifier = {
			modifier = nd_path_of_the_golden_zipa
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mca_formed
			set_variable = nd_mca_path_a
		}
	}
	option = {
		name = nd_mca.1.b
		add_country_modifier = {
			modifier = nd_tundama_defiance
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mca_formed
			set_variable = nd_mca_path_b
		}
	}
}
