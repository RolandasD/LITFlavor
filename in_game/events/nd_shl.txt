namespace = nd_shl

# Formation Event - Up Ewig Ungedeelt
nd_shl.1 = {
	type = country_event
	title = nd_shl.1.t
	desc = nd_shl.1.d
	fire_only_once = yes

	option = {
		name = nd_shl.1.a
		add_country_modifier = {
			modifier = nd_shl_ewig_ungedeelt
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_shl_formed
			set_variable = nd_shl_path_a
		}
	}
	option = {
		name = nd_shl.1.b
		add_country_modifier = {
			modifier = nd_shl_eider_frontier
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = mercantilism_vs_free_trade
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_shl_formed
			set_variable = nd_shl_path_b
		}
	}
}
