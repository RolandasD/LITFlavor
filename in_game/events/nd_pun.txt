namespace = nd_pun

# Formation event — fired from form_effect via trigger_event_non_silently
nd_pun.1 = {
	type = country_event
	title = nd_pun.1.t
	desc = nd_pun.1.d
	fire_only_once = yes

	option = {
		name = nd_pun.1.a
		add_country_modifier = {
			modifier = nd_khalsa_raj
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_pun_formed
			set_variable = nd_pun_path_a
		}
	}
	option = {
		name = nd_pun.1.b
		add_country_modifier = {
			modifier = nd_five_rivers_prosperity
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_pun_formed
			set_variable = nd_pun_path_b
		}
	}
}
