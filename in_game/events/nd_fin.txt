##############################
# Formation Event
##############################

nd_fin.1 = {
	type = country_event
	title = nd_fin.1.t
	desc = nd_fin.1.d
	fire_only_once = yes

	option = {
		name = nd_fin.1.a
		add_country_modifier = {
			modifier = nd_fin_thousand_lakes
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_fin_formed
			set_variable = nd_fin_path_a
		}
	}
	option = {
		name = nd_fin.1.b
		add_country_modifier = {
			modifier = nd_fin_northern_trade
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_fin_formed
			set_variable = nd_fin_path_b
		}
	}
}
