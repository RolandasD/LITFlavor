namespace = nd_nzo

# Formation event — fires when Nanzhao is formed
nd_nzo.1 = {
	type = country_event
	title = nd_nzo.1.t
	desc = nd_nzo.1.d
	fire_only_once = yes

	option = {
		name = nd_nzo.1.a
		add_country_modifier = {
			modifier = nd_tiger_banner_of_the_black_mywa
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_nzo_formed
			set_variable = nd_nzo_path_a
		}
	}
	option = {
		name = nd_nzo.1.b
		add_country_modifier = {
			modifier = nd_southern_silk_road
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_nzo_formed
			set_variable = nd_nzo_path_b
		}
	}
}
