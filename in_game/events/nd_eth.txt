# Formation Event — The Restoration of the Solomonic Throne
nd_eth.1 = {
	type = country_event
	title = nd_eth.1.t
	desc = nd_eth.1.d
	fire_only_once = yes

	option = {
		name = nd_eth.1.a
		add_country_modifier = {
			modifier = nd_sword_of_zion
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_eth_formed
			set_variable = nd_eth_path_a
		}
	}
	option = {
		name = nd_eth.1.b
		add_country_modifier = {
			modifier = nd_light_of_aksum
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_eth_formed
			set_variable = nd_eth_path_b
		}
	}
}
