# Formation Event — The Birth of Livonia
nd_liv.1 = {
	type = country_event
	title = nd_liv.1.t
	desc = nd_liv.1.d
	fire_only_once = yes

	option = {
		name = nd_liv.1.a
		add_country_modifier = {
			modifier = nd_sword_and_cross
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_liv_formed
			set_variable = nd_liv_path_a
		}
	}
	option = {
		name = nd_liv.1.b
		add_country_modifier = {
			modifier = nd_baltic_emporium
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_liv_formed

		}
	}
}
