namespace = nd_spa

# Formation Event — The Birth of Spain
nd_spa.1 = {
	type = country_event
	title = nd_spa.1.t
	desc = nd_spa.1.d
	fire_only_once = yes

	option = {
		name = nd_spa.1.a
		add_country_modifier = {
			modifier = nd_spa_universal_monarchy
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_spa_formed
			set_variable = nd_spa_path_a
		}
	}
	option = {
		name = nd_spa.1.b
		add_country_modifier = {
			modifier = nd_spa_empire_of_the_indies
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_spa_formed
			set_variable = nd_spa_path_b
		}
	}
}
