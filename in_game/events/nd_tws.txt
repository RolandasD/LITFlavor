namespace = nd_tws

# Formation event — fires from form_effect
nd_tws.1 = {
	type = country_event
	title = nd_tws.1.t
	desc = nd_tws.1.d
	fire_only_once = yes

	option = {
		name = nd_tws.1.a
		add_country_modifier = {
			modifier = nd_crown_of_the_mezzogiorno
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_tws_formed
			set_variable = nd_tws_path_a
		}
	}
	option = {
		name = nd_tws.1.b
		add_country_modifier = {
			modifier = nd_crossroads_of_the_mediterranean
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_tws_formed
			set_variable = nd_tws_path_b
		}
	}
}
