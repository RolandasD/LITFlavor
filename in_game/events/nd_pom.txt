namespace = nd_pom

# Formation Event - The Griffin's Banner Rises
nd_pom.1 = {
	type = country_event
	title = nd_pom.1.t
	desc = nd_pom.1.d
	fire_only_once = yes

	option = {
		name = nd_pom.1.a
		add_country_modifier = {
			modifier = nd_pom_griffins_dominion
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_pom_formed
			set_variable = nd_pom_path_a
		}
	}
	option = {
		name = nd_pom.1.b
		add_country_modifier = {
			modifier = nd_pom_wendish_quarter
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = aristocracy_vs_plutocracy
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_pom_formed
			set_variable = nd_pom_path_b
		}
	}
}
