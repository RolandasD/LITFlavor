namespace = nd_tus

# Formation event — The Grand Duchy of Tuscany
nd_tus.1 = {
	type = country_event
	title = nd_tus.1.t
	desc = nd_tus.1.d
	fire_only_once = yes

	option = {
		name = nd_tus.1.a
		add_country_modifier = {
			modifier = nd_tus_palazzo_dei_medici
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_tus_formed
			set_variable = nd_tus_path_a
		}
	}
	option = {
		name = nd_tus.1.b
		add_country_modifier = {
			modifier = nd_tus_porto_franco_spirit
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_tus_formed
			set_variable = nd_tus_path_b
		}
	}
}
