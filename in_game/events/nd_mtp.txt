# Formation Event
##############################

nd_mtp.1 = {
	type = country_event
	title = nd_mtp.1.t
	desc = nd_mtp.1.d
	fire_only_once = yes

	option = {
		name = nd_mtp.1.a
		add_country_modifier = {
			modifier = nd_mtp_golden_trade
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mtp_formed
			set_variable = nd_mtp_path_a
		}
	}
	option = {
		name = nd_mtp.1.b
		add_country_modifier = {
			modifier = nd_mtp_mwari_sovereignty
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mtp_formed

		}
	}
}
