namespace = nd_jap

# Formation event — fires when sengoku ends and JAP is formed via form_country
nd_jap.1 = {
	type = country_event
	title = nd_jap.1.t
	desc = nd_jap.1.d
	fire_only_once = yes

	option = {
		name = nd_jap.1.a
		add_country_modifier = {
			modifier = nd_tenka_fubu
			years = 40
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_jap_formed
			set_variable = nd_jap_path_a
		}
	}
	option = {
		name = nd_jap.1.b
		add_country_modifier = {
			modifier = nd_pax_tokugawa
			years = 40
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_jap_formed
			set_variable = nd_jap_path_b
		}
	}
	option = {
		name = nd_jap.1.c
		add_country_modifier = {
			modifier = nd_land_of_rising_sun
			years = 40
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_jap_formed
			set_variable = nd_jap_path_c
		}
	}
}
