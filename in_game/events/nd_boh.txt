namespace = nd_boh

# Formation event — fires from form_effect via trigger_event_non_silently
nd_boh.1 = {
	type = country_event
	title = nd_boh.1.t
	desc = nd_boh.1.d
	fire_only_once = yes

	option = {
		name = nd_boh.1.a
		add_country_modifier = {
			modifier = nd_boh_hussite_heritage
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_boh_formed
			set_variable = nd_boh_path_a
		}
	}
	option = {
		name = nd_boh.1.b
		add_country_modifier = {
			modifier = nd_boh_crown_wenceslas
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_boh_formed
			set_variable = nd_boh_path_b
		}
	}
}
