namespace = nd_dec

# Formation event — The Deccan Sultanate is proclaimed
nd_dec.1 = {
	type = country_event
	title = nd_dec.1.t
	desc = nd_dec.1.d
	fire_only_once = yes

	option = {
		name = nd_dec.1.a
		add_country_modifier = {
			modifier = nd_dec_sword_of_the_deccan
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_dec_formed
			set_variable = nd_dec_path_a
		}
	}
	option = {
		name = nd_dec.1.b
		add_country_modifier = {
			modifier = nd_dec_golconda_splendor
			years = 30
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_dec_formed
			set_variable = nd_dec_path_b
		}
	}
}
