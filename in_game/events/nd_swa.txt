namespace = nd_swa

# Formation event — the restoration of Swabia
nd_swa.1 = {
	type = country_event
	title = nd_swa.1.t
	desc = nd_swa.1.d
	fire_only_once = yes

	option = {
		name = nd_swa.1.a
		add_country_modifier = {
			modifier = nd_swa_league_of_equals
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_move_to_right
		}
		hidden_effect = {
			set_variable = nd_swa_formed
			set_variable = nd_swa_path_a
		}
	}
	option = {
		name = nd_swa.1.b
		add_country_modifier = {
			modifier = nd_swa_ducal_restoration
			years = 20
			mode = add_and_extend
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_move_to_left
		}
		hidden_effect = {
			set_variable = nd_swa_formed
			set_variable = nd_swa_path_b
		}
	}
}
