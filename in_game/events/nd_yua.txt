namespace = nd_yua

# Event 1: Dayan Khan's Restoration — the Six Tumen Reform
nd_yua.1 = {
	type = country_event
	fire_only_once = yes

	dynamic_historical_event = {
		tag = YUA
		from = 1337.1.1
		to = 1836.1.1
		monthly_chance = 100
	}

	trigger = {
		has_advance = nd_yua_dayan_khan_revival
	}

	title = nd_yua.1.t
	desc = nd_yua.1.d

	option = {
		name = nd_yua.1.a
		add_country_modifier = {
			modifier = nd_yua_six_tumen_order
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_yua_formed
			set_variable = nd_yua_path_a
		}
	}
	option = {
		name = nd_yua.1.b
		add_country_modifier = {
			modifier = nd_yua_borjigin_autonomy
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_yua_formed
			set_variable = nd_yua_path_b
		}
	}
}
