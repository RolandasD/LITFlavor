namespace = nd_mir

# Formation event — fires from form_effect via trigger_event_non_silently
nd_mir.1 = {
	type = country_event
	title = nd_mir.1.t
	desc = nd_mir.1.d
	fire_only_once = yes

	option = {
		name = nd_mir.1.a
		add_country_modifier = {
			modifier = nd_hasina_of_the_highlands
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mir_formed
			set_variable = nd_mir_path_a
		}
	}
	option = {
		name = nd_mir.1.b
		add_country_modifier = {
			modifier = nd_island_of_a_thousand_ports
			years = 20
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_mir_formed
			set_variable = nd_mir_path_b
		}
	}
}
