##############################
# Formation Event
##############################

nd_bej.1 = {
	type = country_event
	title = nd_bej.1.t
	desc = nd_bej.1.d
	fire_only_once = yes

	option = {
		name = nd_bej.1.a
		add_country_modifier = {
			modifier = nd_bej_lords_of_the_red_sea
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_bej_formed
			set_variable = nd_bej_path_a
		}
	}
	option = {
		name = nd_bej.1.b
		add_country_modifier = {
			modifier = nd_bej_sons_of_the_desert
			years = 15
			mode = add_and_extend
		}
		hidden_effect = {
			set_variable = nd_bej_formed
			set_variable = nd_bej_path_b
		}
	}
}
