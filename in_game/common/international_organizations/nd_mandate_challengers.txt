nd_mandate_challengers = {
	has_target = no
	unique = yes
	show_on_diplomatic_map = yes
	expel_members_who_are_targets_of_other_members = no

	has_leader_country = yes
	leader_type = country
	leader_change_method = score
	leader_change_trigger_type = timed
	months_between_leader_changes = 1
	leader_score = {
		value = great_power_score
	}

	leader_modifier = {
		government_size = 1
	}

	can_declare_war = { always = yes }

	create_visible_trigger = { always = no }

	invite_visible_trigger = {
		is_situation_active = situation:nd_mandate_crisis
	}

	can_join_trigger = { always = no }
	can_leave_trigger = { always = no }

	can_lead_trigger = {
		is_subject = no
		is_member_of_international_organization = scope:recipient
	}

	auto_leave_trigger = {
		OR = {
			AND = {
				has_variable = nd_mc_emperor_allegiance
				var:nd_mc_emperor_allegiance >= 50
			}
			NOT = { has_reform = government_reform:warring_state }
			AND = {
				exists = international_organization:middle_kingdom
				is_leader_of_international_organization = international_organization:middle_kingdom
			}
			AND = {
				exists = international_organization:middle_kingdom.leader_country
				is_subject_of = international_organization:middle_kingdom.leader_country
			}
			AND = {
				exists = international_organization:middle_kingdom.leader_country
				is_allied_with = { target = international_organization:middle_kingdom.leader_country }
			}
		}
	}

	auto_disband_trigger = { always = no }

	monthly_effect = {
		if = {
			limit = {
				exists = international_organization:middle_kingdom
				exists = international_organization:middle_kingdom.leader_country
			}
			every_international_organization_member = {
				limit = {
					has_variable = nd_mc_emperor_allegiance
				}
				change_variable = {
					name = nd_mc_emperor_allegiance
					add = {
						value = "opinion(international_organization:middle_kingdom.leader_country)"
						divide = 100
						subtract = 1
					}
				}
				# Clamp to [-100, 100]
				if = {
					limit = { var:nd_mc_emperor_allegiance < -100 }
					set_variable = { name = nd_mc_emperor_allegiance value = -100 }
				}
				if = {
					limit = { var:nd_mc_emperor_allegiance > 100 }
					set_variable = { name = nd_mc_emperor_allegiance value = 100 }
				}
			}
		}
	}

	ai_desire_to_join = {
		add = -1000
	}

	ai_desire_to_allow_new_member = {
		add = 100
	}
}
