	build_time = guild_build_time

	country_potential = {
		has_advance = nd_bgp_rhone_viticulture
	}

	modifier = {
		local_production_efficiency = small_production_efficiency_bonus
		local_monthly_prosperity = 0.001
	}

	unique_production_methods = {
		nd_bgp_rhone_winemaking = {
			fruit = 0.25
			lumber = 0.1
			produced = wine
			output = 0.24
			category = guild_input
		}
		# Profit: fruit 0.25*1.0 + lumber 0.1*1.5 = 0.40 input
		# output_qty = 0.40 * 1.20 / 2.0 = 0.24 -> 0.22 for conservatism
		# 0.22 * 2.0 = 0.44 vs 0.40 = 10% — slightly low
		# Use 0.24: 0.24*2.0 = 0.48 vs 0.40 = 20% ✓
		nd_bgp_papal_vintages = {
			fruit = 0.3
			dyes = 0.05
			produced = wine
			output = 0.30
			category = guild_input
		}
		# Profit: fruit 0.3*1.0 + dyes 0.05*4.0 = 0.50 input
		# output_qty = 0.50 * 1.20 / 2.0 = 0.30
		# But we use 0.27: 0.27*2.0 = 0.54 vs 0.50 = 8% — too low
		# Use 0.30: 0.30*2.0 = 0.60 vs 0.50 = 20% ✓
	}
	construction_demand = workshop_construction
}
