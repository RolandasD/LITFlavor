# Silk Ergasterion — imperial silk workshop producing luxury textiles.
# The Byzantine silk industry was a state monopoly. Ergasteria were the
# workshops where raw silk was spun, dyed, and woven into imperial fabric.
nd_byz_silk_ergasterion = {
	is_special = yes
	is_foreign = no
	pop_type = burghers
	max_levels = 1
	employment_size = guild_employment
	category = consumer_goods_category
	rural_settlement = no
	town = yes
	city = yes
	expensive = yes
	build_time = guild_build_time

	country_potential = {
		has_advance = nd_byz_silk_monopoly
	}

	modifier = {
		local_burghers_estate_power = 0.5
	}

	unique_production_methods = {
		nd_byz_silk_weaving = {
			silk = 0.2
			dyes = 0.1
			produced = fine_cloth
			output = 0.24
			category = guild_input
		}
	}
	# Profit check: silk 0.2x4.0 + dyes 0.1x4.0 = 1.20 input cost
	# output_qty = 1.20 x 1.20 / 6.0 = 0.24 fine_cloth -> profit = 20% (15-30% range)
	construction_demand = guild_construction
}
