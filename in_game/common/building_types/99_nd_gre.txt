	build_time = rural_build_time

	country_potential = {
		has_advance = nd_gre_olive_cultivation
	}

	modifier = {
		local_monthly_food_modifier = 0.05
	}

	unique_production_methods = {
		nd_gre_olive_pressing = {
			olives = 0.3
			livestock = 0.1
			produced = wine
			output = 0.27
			category = guild_input
		}
		nd_gre_currant_export = {
			fruit = 0.4
			produced = liquor
			output = 0.19
			category = guild_input
		}
	}
	# Profit check (nd_gre_olive_pressing):
	# olives 0.3x1.0 + livestock 0.1x1.5 = 0.45 input cost
	# 0.45 x 1.20 / 2.0 = 0.27 wine -> profit = 20% (15-30% range)
	#
	# Profit check (nd_gre_currant_export):
	# fruit 0.4x1.0 = 0.40 input cost
	# 0.40 x 1.20 / 2.5 = 0.192 -> round to 0.19 liquor -> profit = 18.75% (15-30% range)
	construction_demand = village_construction
}
