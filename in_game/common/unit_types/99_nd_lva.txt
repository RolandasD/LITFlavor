# Age 1 — Curonian Raiders (levy infantry)
# The Curonians were known as the "Baltic Vikings" for their seafaring and raiding prowess.
# Fast-moving warriors with high morale damage representing their ferocious reputation.
a_nd_lva_curonian_raider = {
	is_special = yes
	category = army_infantry
	copy_from = a_age_1_traditions_infantry
	buildable = no
	levy = yes

	# Base: max_strength=0.5, combat_power=1
	morale_damage_done = 0.1
	strength_damage_done = 0.1
	movement_speed = 0.5

	combat = {
		forest = 0.1
		wetlands = 0.05
	}

	country_potential = {
		has_or_had_tag = LVA
	}

	gfx_tags = { medium_tag }
}
