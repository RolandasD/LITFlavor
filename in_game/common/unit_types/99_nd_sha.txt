# Shan Highland Cavalry — fast raiders who exploit mountain terrain
# Age 1 cavalry with reduced mountain penalties, representing Shan expertise
# in highland warfare. Used by levies.
a_nd_sha_highland_cavalry = {
	is_special = yes
	category = army_cavalry
	copy_from = a_age_1_traditions_cavalry
	buildable = no
	levy = yes

	# Base template: max_strength=0.2, combat_power=4
	# Mountain penalties: impact mountains 0.10, combat mountains -0.10
	# We reduce the mountain penalty and add flanking
	combat = {
		mountains = -0.05
	}
	impact = {
		mountains = 0.05
	}
	flanking_ability = 0.75

	country_potential = {
		has_or_had_tag = SHA
	}

	gfx_tags = { medium_tag }
}

# Sawbwa's Elephant Guard — war elephants employed by Shan rulers
# Age 3 auxiliary, powerful shock troops requiring elephant goods
# Buildable.
a_nd_sha_elephant_guard = {
	is_special = yes
	category = army_auxiliary
	copy_from = a_age_3_discovery_auxiliary
	buildable = yes

	# Base template: max_strength=0.15, combat_power=0.25
	combat_power = 0.25
	food_storage_per_strength = 11
	impact = {
		jungle = -0.25
	}
	combat = {
		jungle = 0.15
	}

	location_potential = {
		market ?= { is_produced_in_market = goods:elephants }
	}

	country_potential = {
		has_or_had_tag = SHA
	}

	gfx_tags = { elephant_tag auxiliary_tag medium_tag }
	max_strength = 0.165
}
