# Journal Entry: The Grand Duchy Ascendant
# Covers Lithuania's rise to become the largest state in Europe.
######################################
journal_entry = {
    id = lit_je_grand_duchy

    title = lit_je_grand_duchy.t
    desc = lit_je_grand_duchy.d
    picture = "gfx/event_pictures/eu5_military.dds"

    trigger = {
        tag = LIT
    }

    task = {
        id = lit_je_grand_duchy_task_1
        title = lit_je_grand_duchy_task_1.t
        desc = lit_je_grand_duchy_task_1.d

        # Expand to control the major eastern territories
        completed_when = {
            num_of_provinces_owned_or_leased >= 30
        }

        on_completion = {
            add_prestige = 20
            add_country_modifier = {
                name = lit_grand_duchy_power
                duration = 7300
            }
        }
    }

    task = {
        id = lit_je_grand_duchy_task_2
        title = lit_je_grand_duchy_task_2.t
        desc = lit_je_grand_duchy_task_2.d

        # Secure the Baltic coast by controlling provinces in the Baltic region area
        completed_when = {
            any_owned_province = {
                region = baltic_region
            }
        }

        on_completion = {
            add_country_modifier = {
                name = lit_baltic_coast_secured
                duration = 7300
            }
        }
    }

    task = {
        id = lit_je_grand_duchy_task_3
        title = lit_je_grand_duchy_task_3.t
        desc = lit_je_grand_duchy_task_3.d

        # Defeat the Teutonic Order
        completed_when = {
            NOT = { exists = TEU }
        }

        on_completion = {
            add_prestige = 30
            add_country_modifier = {
                name = lit_teutonic_order_defeated
                duration = 10950
            }
        }
    }
}
