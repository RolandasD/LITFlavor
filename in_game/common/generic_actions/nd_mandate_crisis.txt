# Emperor Phase 1 Actions

nd_mc_reform_bureaucracy = {
	type = situation

	ai_tick = monthly
	ai_tick_frequency = 6
	automation_tick = never
	automation_tick_frequency = 12

	potential = {
		scope:actor = {
			is_leader_of_international_organization = international_organization:middle_kingdom
		}
	}

	allow = {
		scope:actor = {
			government_power >= 50
			custom_tooltip = {
				text = nd_mc_has_already_enacted_tt
				NOT = { has_variable = nd_mc_used_reform_bureaucracy }
			}
		}
	}

	select_trigger = {
		looking_for_a = situation
		target_flag = recipient
		name = "nd_mc_choose_situation"
		column = {
			data = name
		}
		visible = {
			situation:nd_mandate_crisis = this
			situation_is_active = yes
		}
	}

	effect = {
		change_celestial_authority = { value = celestial_authority_mild_bonus }
		scope:actor = {
			add_government_power = government_power_mild_penalty
			hidden_effect = { set_variable = nd_mc_used_reform_bureaucracy }
		}
		custom_tooltip = nd_mc_reform_bureaucracy_tt
	}

	ai_will_do = {
		add = 50
		add = {
			if = {
				limit = {
					scope:actor = { government_power >= 80 }
				}
				value = 100
			}
		}
	}
}

# Emperor Phase 2 Actions

nd_mc_negotiate_surrender = {
	type = situation

	ai_tick = monthly
	ai_tick_frequency = 3
	automation_tick = never
	automation_tick_frequency = 12

	price = price:rtr_appease_the_court_price

	potential = {
		scope:actor = {
			is_leader_of_international_organization = international_organization:middle_kingdom
			has_variable = nd_mc_phase
			var:nd_mc_phase = 2
		}
	}

	allow = {
		scope:actor = {
			num_of_diplomats >= 1
		}
	}

	select_trigger = {
		looking_for_a = situation
		target_flag = recipient
		name = "nd_mc_choose_situation"
		column = {
			data = name
		}
		visible = {
			situation:nd_mandate_crisis = this
			situation_is_active = yes
		}
	}

	select_trigger = {
		looking_for_a = country
		target_flag = target
		name = "nd_mc_choose_warlord"
		none_available_msg_key = "nd_mc_no_warlords_available"
		column = {
			data = name
		}
		column = {
			data = nd_mc_emperor_allegiance
		}
		visible = {
			has_variable = nd_mc_emperor_allegiance
			var:nd_mc_emperor_allegiance < 50
			NOT = { is_at_war_with = scope:actor }
			capital = { is_discovered_by = scope:actor }
			within_diplomatic_range = scope:actor
		}
	}

	effect = {
		if = {
			limit = { exists = scope:target }
			scope:actor = {
				add_diplomats = -1
				transfer_yearly_gold = {
					value = 0.5
					target = scope:target
				}
			}
			scope:target = {
				hidden_effect = {
					change_variable = {
						name = nd_mc_emperor_allegiance
						add = 25
					}
				}
				add_opinion = {
					modifier = opinion_rtr_granted_titles
					target = scope:actor
				}
			}
		}
		custom_tooltip = nd_mc_negotiate_surrender_tt
	}

	ai_will_do = {
		add = -50
		add = {
			if = {
				limit = {
					exists = scope:target
					scope:target = {
						var:nd_mc_emperor_allegiance >= 25
						var:nd_mc_emperor_allegiance < 50
					}
				}
				value = 200
			}
		}
		add = {
			if = {
				limit = {
					scope:actor = {
						gold >= {
							value = monthly_income_trade_and_tax
							multiply = 8
						}
					}
				}
				value = 100
			}
		}
	}
}

nd_mc_seek_mandate = {
	type = situation

	ai_tick = monthly
	ai_tick_frequency = 3
	automation_tick = never
	automation_tick_frequency = 12

	potential = {
		scope:actor = {
			has_variable = nd_mc_emperor_allegiance
			is_member_of_international_organization = international_organization:nd_mandate_challengers
		}
		exists = international_organization:middle_kingdom.leader_country
	}

	allow = {
		scope:actor = {
			var:nd_mc_emperor_allegiance < 0
			NOT = { has_variable = nd_mc_claimed_mandate }
			is_subject = no
		}
	}

	select_trigger = {
		looking_for_a = situation
		target_flag = recipient
		name = "nd_mc_choose_situation"
		column = {
			data = name
		}
		visible = {
			situation:nd_mandate_crisis = this
			situation_is_active = yes
		}
	}

	effect = {
		scope:actor = {
			hidden_effect = {
				change_variable = {
					name = nd_mc_emperor_allegiance
					add = -50
				}
			}
			add_prestige = prestige_mild_bonus
			hidden_effect = { set_variable = nd_mc_claimed_mandate }
			if = {
				limit = { NOT = { is_at_war_with = international_organization:middle_kingdom.leader_country } }
				declare_war_with_cb = {
					target = international_organization:middle_kingdom.leader_country
					type = casus_belli:cb_conquer_enemy
				}
			}
		}
		custom_tooltip = nd_mc_seek_mandate_tt
	}

	ai_will_do = {
		add = -100
		add = {
			if = {
				limit = {
					scope:actor = {
						var:nd_mc_emperor_allegiance <= -50
						great_power_score >= 50
					}
				}
				value = 300
			}
		}
		add = {
			if = {
				limit = {
					scope:actor = {
						is_leader_of_international_organization = international_organization:nd_mandate_challengers
					}
				}
				value = 200
			}
		}
	}
}
