REPLACE:IRE_f = {

	level = 2
	required_locations_fraction = 0.75
	rule = historical

	potential = {
		OR = {
			culture = { has_culture_group = culture_group:hibernian_group }
			tag = PLE
		}
	}

	allow = {
		OR = {
			AND = {
				#allow if they're the leader of a high_kingship in the Ireland region
				is_member_of_international_organization_of_type = { type = high_kingship }
				any_international_organizations_member_of = {
					count = all
					filter = {
						international_organization_type = international_organization_type:high_kingship
						has_variable = high_kingship_region
						var:high_kingship_region = region:ireland_region
					}
					leader_country ?= root
				}
			}
			custom_tooltip = {
				text = neither_highkingship_nor_lordship_exist_tt
				NOT = {
					any_international_organization = {
						OR = {
							international_organization_type = international_organization_type:lordship_of_ireland
							international_organization_type = international_organization_type:high_kingship
						}
					}
				}
			}
		}
		custom_tooltip = {
			text = IRE_f_lordship_tt
			not = { is_member_of_international_organization = international_organization:lordship_of_ireland }
		}
	}
	
	name = IRE
	flag = IRE
	adjective = IRE_ADJ
	tag = IRE
	color = map_IRE

	regions = {
		ireland_region
	}
	form_effect = {
		if = {
			limit = { unique_international_organization_type_exists = international_organization_type:lordship_of_ireland }
			destroy_international_organization = { target = international_organization:lordship_of_ireland }
		}
		if = {
			limit = { country_rank_level < 3 }
			set_country_rank_effect = { rank = country_rank:rank_kingdom }
		}
		trigger_event_non_silently = nd_ire.1
		add_stability = stability_mild_bonus
	}
}
