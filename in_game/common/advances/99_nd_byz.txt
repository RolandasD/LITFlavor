# ==========================================
# Byzantium (BYZ) — Custom Advances
# ==========================================
# 17 vanilla advances exist (4 in age 1, 5 in age 2, 2 in age 3,
# 2 in age 4, 2 in age 5, 2 in age 6).
# These 13 advances fill the gaps: naval, economy, administration,
# infrastructure themes that vanilla left untold.

# ------------------------------------------------
# AGE 1 — TRADITIONS
# ------------------------------------------------

# Greek Fire — unlocks the dromon war galley and its levy.
# The most feared weapon in the medieval Mediterranean, Greek fire
# gave Byzantine galleys devastating close-range superiority.
nd_byz_greek_fire = {
	age = age_1_traditions
	potential = { has_or_had_tag = BYZ }
	requires = ship_building_advance
	icon = ship_building_advance
	unlock_levy = levy_nd_byz_dromon
	navy_galley_power = 0.15
	ai_weight = {
		add = "ai_unlock_unit_score(n_nd_byz_dromon)"
	}
}

# Thematic Administration — the theme system gave Byzantium a
# self-sustaining military-administrative structure where generals
# governed provinces and soldiers farmed the land they defended.
nd_byz_thematic_administration = {
	age = age_1_traditions
	potential = { has_or_had_tag = BYZ }
	requires = medieval_administration
	icon = medieval_administration
	global_monthly_control = 0.0015
	global_levy_size_modifier = 0.1
	global_distance_from_capital_speed_propagation = 0.1
}

# ------------------------------------------------
# AGE 2 — RENAISSANCE
# ------------------------------------------------

# Pronoia System — the Byzantine feudal grant system. Pronoiars
# held land in exchange for military service, providing the empire
# with professional cavalry without the costs of standing armies.
nd_byz_pronoia_system = {
	age = age_2_renaissance
	potential = { has_or_had_tag = BYZ }
	requires = drill_army_advance
	icon = drill_army_advance
	army_cavalry_power = 0.1
	army_cavalry_maintenance_cost_modifier = -0.3
	nobles_estate_target_satisfaction = 0.05
}

# Imperial Silk Monopoly — unlocks the silk ergasterion building.
# Byzantine silk was a state monopoly regulated by the Book of the
# Eparch. The imperial workshops produced the finest textiles in
# the world, and Tyrian purple silk was reserved for the emperor.
nd_byz_silk_monopoly = {
	age = age_2_renaissance
	potential = { has_or_had_tag = BYZ }
	requires = merchants_and_trade
	icon = merchants_and_trade
	unlock_building = nd_byz_silk_ergasterion
	global_fine_cloth_output_modifier = 0.15
}

# AGE 4 — REFORMATION
# ------------------------------------------------

# Book of the Eparch — the Byzantine trade regulation manual that
# governed Constantinople's guilds. It specified rules for silk workers,
# perfumers, bakers, butchers, and every other trade — a sophisticated
# regulatory apparatus unmatched in medieval Europe.
nd_byz_book_of_the_eparch = {
	age = age_4_reformation
	potential = { has_or_had_tag = BYZ }
	requires = global_trade_advance
	icon = global_trade_advance
	trade_efficiency = 0.025
	global_merchant_power = 0.1
	global_trade_center_power = 0.15
}

# --- PATH A: HEIR OF ROME ---
# Stage 1: Reclaim the Exarchate — take Ravenna and southern Italy
nd_byz_dest_exarchate = {
	age = age_1_traditions
	requires = nd_byz_destiny
	potential = { tag = BYZ has_variable = nd_byz_formed }
	allow = {
		custom_tooltip = {
			text = nd_byz_path_a_tt
			has_variable = nd_byz_path_a
		}
		owns_or_non_sovereign_subject_owns = location:ravenna
		owns_or_non_sovereign_subject_owns = location:bari
		gold >= 1000
	}
	icon = imperial_ambitions
	land_morale_modifier = 0.1
	siege_ability = 0.1
}

# Stage 2: Reconquista Africana — control the former African provinces
nd_byz_dest_africa = {
	age = age_1_traditions
	requires = nd_byz_dest_exarchate
	potential = { tag = BYZ has_variable = nd_byz_formed }
	allow = {
		owns_or_non_sovereign_subject_owns = location:tunis
		owns_or_non_sovereign_subject_owns = location:tripoli
		army_size >= 50
		prestige >= 50

	icon = feudalism_advance

}

		prestige >= 75
