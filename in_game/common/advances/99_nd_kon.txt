	age = age_1_traditions
	requires = nd_kon_destiny
	potential = { tag = KON has_variable = nd_kon_formed }
	allow = {
		custom_tooltip = {
			text = nd_kon_path_a_tt
			has_variable = nd_kon_path_a
		}
		has_embraced_institution = institution:banking
		any_market_present_in_country = {
			OR = {
				is_produced_in_market = goods:ivory
				is_traded_in_market = goods:ivory
			}
		}
		gold >= 1000
	}
	icon = diplomatic_training
	legislative_efficiency = 0.05
	global_integration_speed_modifier = 0.1
}
