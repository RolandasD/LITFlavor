### Path A — Defender of the North ###

# Stage 1: Secure the Karelian borderlands

	age = age_1_traditions
	requires = nd_fin_destiny
	potential = { tag = FIN has_variable = nd_fin_formed }
	allow = {
		custom_tooltip = {
			text = nd_fin_path_a_tt
			has_variable = nd_fin_path_a
		}
		stability >= 75
		owns = location:korela
		army_size >= 30
	}
	icon = feudalism_advance
	siege_ability = 0.1
	global_supply_limit_modifier = 0.1
}

# Stage 2: Expand into Aunus Karelia and consolidate
nd_fin_dest_shield_of_north = {
	age = age_1_traditions

	potential = { tag = FIN has_variable = nd_fin_formed }
	allow = {
		owns = location:olonets
		army_size >= 50
		has_embraced_institution = institution:professional_armies
		total_development >= 1000
	}
	icon = imperial_ambitions
	army_movement_speed = 0.1
	global_separatism = -0.05
}
