# Path A: Padshah-e-Hind — military conquest beyond the subcontinent
# Stage 1: Secure the Northwest Frontier (Kabul + Kandahar)
nd_hin_dest_frontier = {
	age = age_1_traditions
	requires = nd_hin_destiny
	potential = { tag = HIN has_variable = nd_hin_formed }
	allow = {
		custom_tooltip = {
			text = nd_hin_path_a_tt
			has_variable = nd_hin_path_a
		}
		owns_or_non_sovereign_subject_owns = location:kabul
		owns_or_non_sovereign_subject_owns = location:kandahar
		army_size >= 30
	}
	icon = feudalism_advance
	siege_ability = 0.1
	army_movement_speed = 0.1
}

# Stage 2: Conquer Persia
nd_hin_dest_persia = {
	age = age_1_traditions
	requires = nd_hin_dest_frontier
	potential = { tag = HIN has_variable = nd_hin_formed }
	allow = {
		owns_or_non_sovereign_subject_owns = location:isfahan
		owns_or_non_sovereign_subject_owns = location:herat
		army_size >= 50
		prestige >= 50
	}
	icon = imperial_ambitions
	global_mercenaries_modifier = 0.2
	global_levy_size_modifier = 0.1
}

# Stage 3: Imperium — reach Baghdad and Mecca, the apex of universal empire
nd_hin_dest_imperium = {
	age = age_1_traditions
	requires = nd_hin_dest_persia
	potential = { tag = HIN has_variable = nd_hin_formed }
	allow = {
		owns_or_non_sovereign_subject_owns = location:baghdad
		owns_or_non_sovereign_subject_owns = location:mecca

		gold >= 3000
	}

}

# Stage 2: Institutional development and silk trade
nd_hin_dest_prosperity = {
	age = age_1_traditions
	requires = nd_hin_dest_ports
	potential = { tag = HIN has_variable = nd_hin_formed }
	allow = {
		has_embraced_institution = institution:banking
		gold >= 2000
		any_market_present_in_country = {
			OR = {
				is_produced_in_market = goods:silk

			}
		}
	}
	icon = global_trade_advance
	global_monthly_prosperity = 0.001
	global_institution_growth_modifier = 0.15
