# Stage 1: The United Kingdom — consolidate the home islands into an economic union

	age = age_1_traditions
	requires = nd_gbr_destiny
	potential = { tag = GBR has_variable = nd_gbr_formed }
	allow = {
		custom_tooltip = {
			text = nd_gbr_path_b_tt
			has_variable = nd_gbr_path_b
		}
		owns_or_non_sovereign_subject_owns = location:newcastle
		stability >= 75
		any_market_present_in_country = {
			OR = {
				is_produced_in_market = goods:coal
				is_traded_in_market = goods:coal
			}
		}
	}
	icon = agriculture_advance
	country_cabinet_efficiency = 0.05
	global_supply_limit_modifier = 0.075
}
